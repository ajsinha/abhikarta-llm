Quick Configuration

# config/application.properties

# Default Provider
llm.default_provider=openai

# OpenAI
llm.openai.enabled=true
llm.openai.api_key=${OPENAI_API_KEY}
llm.openai.default_model=gpt-4o

# Anthropic
llm.anthropic.enabled=true
llm.anthropic.api_key=${ANTHROPIC_API_KEY}
llm.anthropic.default_model=claude-3-5-sonnet-20241022

# Google
llm.google.enabled=true
llm.google.api_key=${GOOGLE_API_KEY}

# Hugging Face
llm.huggingface.enabled=true
llm.huggingface.api_key=${HUGGINGFACE_API_KEY}

# Ollama (Local)
llm.ollama.enabled=true
llm.ollama.base_url=http://localhost:11434

Fallback Chain

Configure automatic failover between providers:

# If OpenAI fails, try Anthropic, then Ollama
llm.fallback_chain=openai,anthropic,ollama
