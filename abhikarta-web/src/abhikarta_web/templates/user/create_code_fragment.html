Example Code 
"""
Data validation utilities.
"""

def validate_email(email: str) -> bool:
    import re
    pattern = r'^[\w\.-]+@[\w\.-]+\.\w+$'
    return bool(re.match(pattern, email))

def validate_required(data: dict, fields: list) -> list:
    missing = []
    for field in fields:
        if not data.get(field):
            missing.append(field)
    return missing
